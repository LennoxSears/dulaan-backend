üíª Integration Code Example

1. Basic Setup

// Import the optimized voice control import { OptimizedAIVoiceControl } from './client/modes/optimized-ai-voice-control.js'; // Create instance with your API endpoint const voiceControl = new OptimizedAIVoiceControl({ apiEndpoint: 'https://your-api-endpoint.com', responseTimeout: 3000, conversationTimeout: 30000 });

2. Start Voice Control

// Set up callbacks for responses voiceControl.setCallbacks({ onResponse: (result) => { console.log('Transcription:', result.transcription); console.log('PWM Value:', result.newPwmValue); console.log('Response:', result.response); // Update your motor control updateMotorPWM(result.newPwmValue); }, onError: (error) => { console.error('Voice control error:', error); }, onVoiceStateChange: (state) => { console.log('Voice state:', state.isActive ? 'Listening' : 'Silent'); } }); // Start the voice control system await voiceControl.start();

3. Monitor Efficiency

// Get real-time efficiency statistics const stats = voiceControl.getEfficiencyStats(); console.log('API Efficiency:', stats.processor.efficiencyPercentage + '%'); console.log('Total API Calls:', stats.api.totalApiCalls); console.log('Speech Segments:', stats.processor.speechChunksSent); // Update UI with stats updateEfficiencyDisplay(stats);

4. Advanced Configuration

// Customize VAD parameters for your environment const customConfig = { vadThresholds: { energy: 0.008, // Energy threshold zcr: 0.08, // Zero crossing rate voiceFrames: 3, // Frames to confirm voice silenceFrames: 20 // Frames to confirm silence }, bufferSizes: { speechBuffer: 30, // 30 seconds max speech vadBuffer: 0.3, // 300ms context buffer preSpeech: 0.3, // 300ms pre-speech context postSpeech: 0.2 // 200ms post-speech context } }; const voiceControl = new OptimizedAIVoiceControl(customConfig);
